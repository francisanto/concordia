  const allGroups = JSON.parse(localStorage.getItem('concordia-groups') || '[]')
  
  for (const group of allGroups) {
    const groupCode = localStorage.getItem(`group_code_${group.groupId || group.id}`)
    if (groupCode === code) {
      return group.groupId || group.id
    }
  }
